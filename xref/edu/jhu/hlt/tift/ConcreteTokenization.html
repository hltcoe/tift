62      /**
63       * Generate a {@link Tokenization} object from a list of tokens,
64       * list of offsets, and start position of the text (e.g., first text
65       * character in the text).
66       * 
67       * @param tokens
68       *            - a {@link List} of tokens (Strings)
69       * @param offsets
70       *            - an array of integers (offsets)
71       * @param startPos
72       *            - starting position of the text
73       * @return a {@link Tokenization} object with correct tokenization
74       */
75      public static Concrete.Tokenization generateConcreteTokenization(List<String> tokens, int[] offsets, int startPos) {
76          Concrete.Tokenization.Builder tokenizationBuilder = Concrete.Tokenization.newBuilder().setUuid(IdUtil.generateUUID())
77                  .setKind(Concrete.Tokenization.Kind.TOKEN_LIST);
78          // Note: we use token index as token id.
79          for (int tokenId = 0; tokenId < tokens.size(); ++tokenId) {
80              String token = tokens.get(tokenId);
81              int start = startPos + offsets[tokenId];
82              int end = start + token.length();
83              tokenizationBuilder.addTokenBuilder().setTokenId(tokenId).setText(token)
84                      .setTextSpan(Concrete.TextSpan.newBuilder().setStart(start).setEnd(end));
85          }
86  
87          return tokenizationBuilder.build();
88      }

108     /**
109      * Generate a {@link Tokenization} object from a list of tokens,
110      * list of tags, list of offsets, and start position of the text (e.g., first text
111      * character in the text). Assumes tags are part of speech tags.
112      * 
113      * Invokes {@link #generateConcreteTokenization(List, int[], int)}
114      * then adds tagging.
115      * 
116      * @see #generateConcreteTokenization(List, int[], int)
117      * 
118      * @param tokens
119      *            - a {@link List} of tokens (Strings)
120      * @param offsets
121      *            - an array of integers (offsets)
122      * @param startPos
123      *            - starting position of the text
124      * @return a {@link Tokenization} object with correct tokenization and token tagging
125      */
126     public static Concrete.Tokenization generateConcreteTokenization(List<String> tokens, List<String> tokenTags, 
127             int[] offsets, int startPos) {
128         Concrete.Tokenization tokenization = generateConcreteTokenization(tokens, offsets, startPos);
129         TokenTagging.Builder ttBuilder = TokenTagging.newBuilder()
130                 .setUuid(IdUtil.generateUUID())
131                 .setMetadata(tiftMetadata);
132         for (int i = 0; i < tokens.size(); i++) {
133             String tag = tokenTags.get(i);
134             if (tag != null) {
135                 TaggedToken tt = TaggedToken.newBuilder()
136                     .setTokenId(offsets[i])
137                     .setTag(tokenTags.get(i))
138                     .build();
139                 ttBuilder.addTaggedToken(tt);
140             }
141         }
142         
143         Concrete.Tokenization.Builder builder = Tokenization.newBuilder(tokenization)
144                 .addPosTags(ttBuilder);
145         return builder.build();
146     }

148     public static Concrete.Tokenization generateConcreteTokenization(TaggedTokenizationOutput tto) {
149         return generateConcreteTokenization(tto.getTokens(), tto.getTokenTags(), convertIntegers(tto.getOffsets()), 0);
150     }

152     /**
153      * Convert a {@link List} of {@link Integer} objects to an integer array primitive.
154      * 
155      * Will throw a {@link NullPointerException} if any element in the list is null. 
156      * 
157      * @param integers a {@link List} of {@link Integer} objects, none of which are <code>null</code>
158      * @return a primitive array of ints
159      */
160     public static int[] convertIntegers(List<Integer> integers) {
161         int[] ret = new int[integers.size()];
162         Iterator<Integer> iterator = integers.iterator();
163         for (int i = 0; i < ret.length; i++) 
164             ret[i] = iterator.next().intValue();
165         
166         return ret;
167     }
