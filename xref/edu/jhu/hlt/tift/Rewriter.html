134     /**
135      * A conservative version of the PTB patterns, meant to (hopefully) be
136      * portable across formal/informal Western (?) languages.
137      */
138     public static Set<PatternStringTuple> getBasicPatterns() {
139         // cut-n-paste, then modified from getPTBPatterns
140         String[] v = {
141                 // double quotes
142                 "([\"\u201c\u201d\u201e\u201f\u275d\u275e])", " $1 ",
143 
144                 // Ellipsis
145                 "\\.\\.\\.", " ... ",
146 
147                 "([,;:@#$%&\\*])", " $1 ",
148 
149                 // HTML escaped (stop gap)
150                 "& ([gl])t ;", "&$1t;",
151                 "& nbsp ;", " &nbsp; ",
152                 "& hearts ;", " &hearts; ",
153 
154                 // vandurme: carefully with final .
155                 "([^\\.])(\\.)(\\s|$)", "$1 $2$3",
156 
157 
158                 // however, we may as well split ALL question marks and exclamation
159                 // points, since they shouldn't have the abbrev.-marker ambiguity
160                 // problem.
161                 //"([\\?!])", " $1 ",
162                 // vandurme> adding unicode characters
163                 // \u00a1 : ! inverted
164                 // \u00bf : ? inverted
165                 "([\\?!\u00a1\u00bf])", " $1 ",
166 
167                 // parentheses, brackets, etc.
168                 "([\\]\\[\\(\\){}<>])", " $1 ",
169 
170                 "--", " -- "
171         };
172 
173         return convertStringArrayPatternsToTupleSet(v);
174     }

176     /**
177      * Based on inspection of:
178      * 
179      * http://www.cis.upenn.edu/~treebank/tokenizer.sed
180      * 
181      * The header of which identifies the author as:
182      * "Robert MacIntyre, University of Pennsylvania, late 1995".
183      */
184     public static Set<PatternStringTuple> getPTBPatterns() {
185         // The following is a port of patterns and comments from tokenizer.sed
186         String[] v = {
187                 // attempt to get correct forward directional quotes, close quotes
188                 // handled at end
189                 "^\"", "`` ",
190                 "([ \\(\\[{<])\"", "$1 `` ",
191 
192                 "\\.\\.\\.", "...",
193                 "([,;:@#$%&])", " $1 ",
194 
195                 // Assume sentence tokenization has been done first, so split FINAL
196                 // periods only. (vandurme: WARNING this is often not true for us)
197                 "([^\\.])([\\.])([\\]\\)}>\"']*) *$", "$1 $2$3 ",
198 
199                 // however, we may as well split ALL question marks and exclamation
200                 // points, since they shouldn't have the abbrev.-marker ambiguity
201                 // problem
202                 "([\\?!])", " $1 ",
203 
204                 // parentheses, brackets, etc.
205                 "([\\]\\[\\(\\){}<>])", " $1 ",
206 
207                 "--", " -- ",
208 
209                 // NOTE THAT SPLIT WORDS ARE NOT MARKED. Obviously this isn't great,
210                 // since you might someday want to know how the words originally fit
211                 // together -- but it's too late to make a better system now, given
212                 // the millions of words we've already done "wrong".
213 
214                 // First off, add a space to the beginning and end of each line, to reduce
215                 // necessary number of regexps.
216                 "$", " ",
217                 "^", " ",
218 
219                 // (vandurme: this is the closing quotation MacIntyre refers to earlier)
220                 "\"", " '' ",
221 
222                 // possessive or close-single-quote
223                 "([^'])' ", "$1 ' ",
224 
225                 // as in it's, I'm, we'd
226                 "'([sSmMdD])", " '$1 ",
227 
228                 "'ll ", " 'll ",
229                 "'re ", " 're ",
230                 "'ve ", " 've ",
231                 "n't ", " n't ",
232                 "'LL ", " 'LL ",
233                 "'RE ", " 'RE ",
234                 "'VE ", " 'VE ",
235                 "N'T ", " N'T ",
236 
237                 " ([Cc])annot ", " $1an not",
238                 " ([Dd])'ye ", " $1' ye",
239                 " ([Gg])imme ", " $1im me ",
240                 " ([Gg])onna ", " $1on na ",
241                 " ([Gg])otta ", " $1ot ta ",
242                 " ([Ll])emme ", " $1em me ",
243                 " ([Mm])ore'n ", " $1ore 'n ",
244                 " ('[Tt])is ", " $1 is ",
245                 " ('[Tt])was ", " $1 was ",
246                 " ([Ww])anna ", " $1an na ",
247                 //" ([Ww])haddya ", " $1ha dd ya ",
248                 //" ([Ww]hatcha ", " $1ha t cha ",
249 
250                 // clean out extra spaces
251                 " +", " ",
252                 "^ +", ""
253         };
254 
255         return convertStringArrayPatternsToTupleSet(v);
256     }

258     private static Set<PatternStringTuple> convertStringArrayPatternsToTupleSet(String[] patternArray) {
259         Set<PatternStringTuple> patterns = new HashSet<PatternStringTuple>(patternArray.length);
260         for (int i = 0; i < patternArray.length - 1; i += 2) 
261             patterns.add(new PatternStringTuple(Pattern.compile(patternArray[i], Pattern.MULTILINE), patternArray[i + 1]));
262 
263         return Collections.unmodifiableSet(patterns);
264     }
265 
266     private static String rewrite(String text, Set<PatternStringTuple> patterns) {
267         String x = text;
268         for (PatternStringTuple pair : patterns)
269             x = pair.getPattern().matcher(x).replaceAll(pair.getEntry());
270 
271         return x.trim();
272     }
