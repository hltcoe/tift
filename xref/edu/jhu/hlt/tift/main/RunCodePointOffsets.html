35      /**
36       * This function takes 2 arguments:
37       * 
38       * <pre>
39       * 1) path to a file of text, and
40       * 2) path to a file of tokens, one per line
41       * </pre>
42       * 
43       * This function reads the text into a string, reads the tokens into a List
44       * of Strings, then outputs code point offsets for the text-token
45       * combination into a file called "output.txt".
46       * 
47       * @param args
48       *            - length 2 array with a path to the original text, and a path
49       *            to token file, one per line
50       * @throws IOException
51       */
52      public static void main(String[] args) throws IOException {
53          if (args.length != 2) {
54              logger.info("Usage: RunCodePointOffsets [path/to/original/text] [path/to/tokens]");
55              System.exit(1);
56          }
57  
58          File inputTextFile = new File(args[0]);
59          File inputTokenFile = new File(args[1]);
60  
61          String text = "";
62          Scanner sc = new Scanner(inputTextFile);
63          while (sc.hasNextLine())
64              text += sc.nextLine();
65          sc.close();
66  
67          List<String> tokenList = new ArrayList<>();
68          sc = new Scanner(inputTokenFile);
69          while (sc.hasNextLine())
70              tokenList.add(sc.nextLine());
71          sc.close();
72  
73          int[] codePointOffsets = Tokenizer.getOffsets(text, tokenList);
74          FileWriter fw = new FileWriter(new File("output.txt"));
75          for (int i = 0; i < codePointOffsets.length; i++)
76              fw.write(codePointOffsets[i] + "\n");
77  
78          fw.close();
79      }
