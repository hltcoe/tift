35      /**
36       * @param args
37       * @throws IOException 
38       */
39      public static void main(String... args) throws IOException {
40          if (args.length != 4) {
41              logger.info("Usage: TextToConcreteFile [tokenization_type] [path/to/text/file] [position_of_first_nonspace_char] " +
42                          "[path/to/output/file]");
43          }
44          
45          int startPos;
46          try {
47              startPos = Integer.parseInt(args[2]);
48          } catch (NumberFormatException nfe) {
49              throw new RuntimeException("3rd argument must be an integer; was: " + args[2], nfe);
50          }
51          
52          String text = "";
53          File input = new File(args[1]);
54          Scanner sc = new Scanner(input);
55          while (sc.hasNextLine())
56              text += sc.nextLine();
57          sc.close();
58  
59          Communication comm = Tokenizer.generateCommunicationWithSingleTokenization("chris_spanish_docs", 
60                  "chris_spanish_docs_" + input.getName(), Tokenizer.BASIC, text, startPos);
61          
62          ProtocolBufferWriter pbw = new ProtocolBufferWriter(new FileOutputStream(new File(args[3])));
63          
64          pbw.write(comm);
65          pbw.close();
66      }
