91      private static void initializePatterns() throws IOException {
92          // email before URL, as they both things like ".com", but email has
93          // '@' email before mention, as anything not email with an '@' is
94          // likely to be a mention
95          tupleList.add(getEmailPatterns());
96          tupleList.add(getMentionPatterns());
97          tupleList.addAll(getURLPatterns());
98          tupleList.add(getWesternEmoticonPatterns());
99          tupleList.add(getEasternEmoticonPatterns());

103         tupleList.add(getLeftArrowPatterns());
104         tupleList.add(getRightArrowPatterns());
105         tupleList.addAll(getRepeatedPatterns());
106         tupleList.addAll(getUnicodePatterns());
107         tupleList.add(getNumberPatterns());

110     public static List<PatternStringTuple> getURLPatterns() {
111         // "p:" gets picked up by the emoticon pattern, so order of patterns is
112         // important. Matching <, > pairs without verifying both are present.
113         List<PatternStringTuple> tupleList = new ArrayList<>();
114         tupleList.add(new PatternStringTuple(START + "(" + "(https?:|www\\.)\\S+" + "|" +
115         // inspired by twokenize
116                 "[^\\s@]+\\.(com|co\\.uk|org|net|info|ca|ly|mp|edu|gov)(/(\\S*))?" + ")" + END, "URL"));
117         tupleList.add(new PatternStringTuple("(?<=\\(|<)" + "(" + "(https?:|www\\.)\\S+" + "|" +
118         // inspired by twokenize
119                 "[^\\s@]+\\.(com|co\\.uk|org|net|info|ca|ly|mp|edu|gov)(/(\\S*))?" + ")" + "(?=\\)|>)", "URL"));
120         return tupleList;

124     // emoticons: (here just for misc reference, not all nec. supported)
125     // http://www.urbandictionary.com/define.php?term=emoticon
126     //
127     // :) smile
128     // :( frown
129     // ;) wink
130     // :P or :
131     // Public tongue sticking out: joke, sarcasm or disgusting
132     // 8) has sunglasses: looking cool
133     // :O surprised
134     // :S confused
135     // :'( shedding a tear
136     // XD laughing, eyes shut (LOL)
137     // XP Tongue out, eyes shut
138     // ^_^ smiley
139     // ^.^ see above, but rather than a wide, closed mouth, a small mouth is
140     // present
141     // ^_~ wink
142     // >_< angry, frustrated
143     // =_= bored
144     // -_- annoyed
145     // -_-' or ^_^' or ^_^;; nervousness, sweatdrop or embarrassed.
146     //
147     // I have observed :3 as semi-frequent, but could be either emoticon, or,
148     // e.g.: 2:30
149     public static PatternStringTuple getWesternEmoticonPatterns() {
150         // Light modification of Potts
151 
152         String eyebrows = "[<>]";
153         String eyes = "[:;=8xX]";
154         String nose = "[\\-oO\\*\\']";
155         // * can be a nose: :*)
156         // or a mouth, for "kisses" : :*
157         String mouth = "[\\*\\)\\]\\(\\[$sSdDpP/\\}\\{@\\|\\\\]";
158 
159         return new PatternStringTuple(START + "((" + eyebrows + "?" + eyes + nose + "?" + mouth + "+" + ")|(" +
160         // reverse
161                 mouth + "+" + nose + "?" + eyes + eyebrows + "?" + "))" + END, "WEST_EMOTICON");
162     }

164     public static PatternStringTuple getEasternEmoticonPatterns() {
165         return new PatternStringTuple(START + "((-_-)|(\\^_\\^)|(=_=)|(\\^\\.\\^)|(\\._\\.)|(>_<)|(\\*-\\*)|(\\*_\\*))" + END,
166                 "EAST_EMOTICON");
167     }

174     public static PatternStringTuple getPhoneNumberPatterns() {
175         // From Potts
176 
177         // Phone numbers:
178         // (?:
179         // (?: # (international)
180         // \+?[01]
181         // [\-\s.]*
182         // )?
183         // (?: # (area code)
184         // [\(]?
185         // \d{3}
186         // [\-\s.\)]*
187         // )?
188         // \d{3} # exchange
189         // [\-\s.]*
190         // \d{4} # base
191         return new PatternStringTuple(START + "((\\+?[01][\\-\\s.]*)?([\\(]?\\d{3}[\\-\\s.\\)]*)?\\d{3}[\\-\\s.]*\\d{4})" + END,
192                 "PhoneNumber");
193     }

210     public static PatternStringTuple getHashtagPatterns() {
211         // Potts: "(\\#+[\\w_]+[\\w\\'_\\-]*[\\w_]+)"
212         // twokenize: #[a-zA-Z0-9_]+
213         // comment from twokenize:
214         // "also gets #1 #40 which probably aren't hashtags .. but good as tokens"
215         return new PatternStringTuple(START + "(\\#+[\\w_]+[\\w\\'_\\-]*[\\w_]+)" + END, "HASHTAG");
216     }

234         // modified from twokenize
235         return new PatternStringTuple(START_W_PAREN +
236         // added the [^.] guard: much more likely to catch punctuation ahead of

238         // @-mention then an email address that ends in '.'
239         // That guard also requires email address to be at least 2 characters
240         // long
241                 "([a-zA-Z0-9\\._%+-]+[^\\.\\!\\?\\:\\;\\s]@[a-zA-Z0-9.-]+\\.[a-zA-Z]{2,4})" + END_W_PAREN, "EMAIL");

299     public static List<PatternStringTuple> getUnicodePatterns() throws IOException {
300         List<PatternStringTuple> tupleList = new ArrayList<>();
301         BufferedReader reader = new BufferedReader(new InputStreamReader(TwitterTokenizer.class.getClassLoader()
302                 .getResourceAsStream("unicode.csv")));
303         String line;
304         String[] toks;
305         char[] pair;
306         String regexp;
307         while ((line = reader.readLine()) != null) {
308             toks = line.split(",");
309             if (toks[0].length() > 4) {
310                 pair = Character.toChars(Integer.decode("0x" + toks[0]));
311                 regexp = "(" + pair[0] + pair[1] + ")";
312                 tupleList.add(new PatternStringTuple(Pattern.compile(regexp), toks[1]));
313             }
314             regexp = "(\\u" + toks[0] + ")";
315             tupleList.add(new PatternStringTuple(Pattern.compile(regexp), toks[1]));
316         }
317         
318         reader.close();
319

330      * Returns 3 arrays:

332      * tokenization

336     public static String[][] tokenizeToArray(String text) {
337         List<TokenTagTuple> x = recursiveTokenize(text.trim(), 0, Tokenizer.BASIC);
338 
339         String[][] y = new String[3][];
340         y[0] = new String[x.size()];
341         y[1] = new String[x.size()];
342         y[2] = new String[x.size()];
343 
344         for (int i = 0; i < x.size(); i++) {
345             y[0][i] = x.get(i).getToken();
346             y[1][i] = x.get(i).getTag();
347         }
348         int[] z = Tokenizer.getOffsets(text, y[0]);
349         for (int i = 0; i < z.length; i++)
350             y[2][i] = "" + z[i];
351 
352         return y;
353     }

363     public static String[] tokenizeTweet(String text, Tokenizer tokenization) {
364         List<TokenTagTuple> x = recursiveTokenize(text.trim(), 0, tokenization);
365 
366         String[] y = new String[x.size()];
367         if (!rw) {
368             for (int i = 0; i < x.size(); i++)
369                 y[i] = x.get(i).getToken();
370         } else {
371             for (int i = 0; i < x.size(); i++) {
372                 if (x.get(i).getTag() != null)
373                     y[i] = "[" + x.get(i).getTag() + "]";
374                 else
375                     y[i] = x.get(i).getToken();
376             }
377         }
378         return y;
379     }

381     private static List<TokenTagTuple> recursiveTokenize(String text, int index, Tokenizer tokenization) {
382         if (index < tupleList.size()) {
383             PatternStringTuple pst = tupleList.get(index);
384             Pattern pattern = pst.getPattern();
385             String tag = pst.getEntry();
386             Matcher matcher;
387             matcher = pattern.matcher(text);
388             int groupCount = matcher.groupCount();
389             String textFragment = "";
390             if (groupCount > 0) {
391                 List<List<TokenTagTuple>> arrays = new ArrayList<>();
392                 int lastEnd = 0;
393                 while (matcher.find()) {
394                     if (matcher.start() > lastEnd) {
395                         textFragment = text.substring(lastEnd, matcher.start()).trim();
396                         if (textFragment.length() > 0) // possible could have
397                                                        // started all as
398                                                        // whitespace
399                             arrays.add(recursiveTokenize(textFragment, index + 1, tokenization));
400                     }
401                     // System.out.println("[" + matcher.group() + "] " +
402                     // matcher.start() + " " + matcher.end());
403                     List<TokenTagTuple> tmpList = new ArrayList<>();
404                     tmpList.add(new TokenTagTuple(matcher.group(), tag));
405                     arrays.add(tmpList);
406                     lastEnd = matcher.end();
407                 }
408                 if (lastEnd < text.length())
409                     arrays.add(recursiveTokenize(text.substring(lastEnd, text.length()).trim(), index + 1, tokenization));
410 
411                 return concatAll(arrays);
412             } else {
413                 return recursiveTokenize(text.trim(), index + 1, tokenization);
414             }
415         } else {
416             List<String> tokenList = tokenization.tokenize(text);
417             List<TokenTagTuple> y = new ArrayList<>(tokenList.size());
418             for (String token : tokenList)
419                 y.add(new TokenTagTuple(token, null));
420             return y;
421         }
422     }

424     public static List<TokenTagTuple> concatAll(List<List<TokenTagTuple>> arrays) {
425         int totalLength = 0;
426         for (List<TokenTagTuple> array : arrays)
427             totalLength += array.size();
428         
429         List<TokenTagTuple> result = new ArrayList<>(totalLength);
430         for (List<TokenTagTuple> array : arrays)
431             result.addAll(array);
432         
433         return result;
434     }
